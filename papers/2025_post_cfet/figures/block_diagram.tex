% figures/block_diagram.tex
\begin{tikzpicture}[node distance=10mm, >=Stealth]
% ---- local styles (self-contained) ----
\tikzset{
  blk/.style={draw, rounded corners=2pt, minimum width=20mm,
              minimum height=6mm, align=center, font=\footnotesize,
              inner sep=2pt, fill=blue!10},
  io/.style={blk, fill=gray!20},
  proc/.style={blk, fill=green!15, minimum width=24mm},
  link/.style={-Stealth, thick}
}

% ---- blocks ----
\node[io]   (src)   {Source};
\node[proc, right=8mm of src] (fe)  {Front-End};
\node[proc, right=8mm of fe]  (adc) {ADC};
\node[proc, right=8mm of adc] (dsp) {DSP};
\node[io,  right=8mm of dsp] (sink) {Output};

% ---- connections ----
\draw[link] (src) -- (fe);
\draw[link] (fe) -- (adc);
\draw[link] (adc) -- (dsp);
\draw[link] (dsp) -- (sink);
\end{tikzpicture}
